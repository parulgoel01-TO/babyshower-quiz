function genCode() {
const c = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
return Array.from({length:5}, () => c[Math.floor(Math.random()*c.length)]).join('');
}

function shuffle(arr) {
const a = [...arr];
for (let i = a.length-1; i > 0; i--) { const j = Math.floor(Math.random()*(i+1)); [a[i],a[j]]=[a[j],a[i]]; }
return a;
}

async function joinRoom(code, name) {
const roomRef = ref(db, `rooms/${code}`);
const snap = await get(roomRef);
if (!snap.exists()) { err('home-err', 'Room not found — check the code'); return; }
const room = snap.val();
if (room.status === 'playing') { err('home-err', 'Quiz already in progress'); return; }
await set(ref(db, `rooms/${code}/players/${name}`), { score: 0, correct: 0, total: 0, avgTime: 0, joined: Date.now() });
G.timeLimit = room.timeLimit;
G.questions = room.questions;
renderLobby(code, false);
showScreen('screen-lobby');
listenForStart(code);
}

window.createRoom = async function() {
const qSetVal = document.getElementById('q-set').value;
let pool;
if (qSetVal === 'custom') {
try { pool = JSON.parse(document.getElementById('custom-json').value); if (!pool[0].q) throw 0; }
catch { err('setup-err','Invalid JSON — check format'); return; }
} else {
pool = shuffle(QUESTION_SETS[qSetVal]);
}
const maxQ = parseInt(document.getElementById('q-count').value);
const questions = pool.slice(0, maxQ);
const timeLimit = parseInt(document.getElementById('time-slider').value);
const code = genCode();
G.roomCode = code;
G.isHost = true;
G.timeLimit = timeLimit;
G.questions = questions;
G.currentQ = 0;
await set(ref(db, `rooms/${code}`), {
status: 'lobby',

questions,
currentQ: 0,
phase: 'lobby',
host: G.playerName,
created: Date.now()
});
await set(ref(db, `rooms/${code}/players/${G.playerName}`), { score:0, correct:0, total:0, avgTime:0, joined:Date.now() });
renderLobby(code, true);
showScreen('screen-lobby');
listenLobbyPlayers(code);
};

function listenForStart(code) {

if (snap.val() === 'playing') {

G.questions = room.questions;
G.timeLimit = room.timeLimit;
G.currentQ = room.currentQ || 0;
clearListeners();
loadQuestion();
showScreen('screen-question');
listenForPhase(code);
}
});
G.listeners.push({ r, unsub });
}

function clearListeners() {
G.listeners.forEach(({ r, unsub }) => { try { off(r); } catch(e){} });
G.listeners = [];
}

window.startQuiz = async function() {

loadQuestion();
showScreen('screen-question');
listenForPhase(G.roomCode);
};

function listenForPhase(code) {
const r = ref(db, `rooms/${code}`);
const unsub = onValue(r, snap => {
if (!snap.exists()) return;
const room = snap.val();
if (G.isHost) return;
const newQ = room.currentQ || 0;
const phase = room.phase;
if (phase === 'question' && newQ !== G.currentQ) {
G.currentQ = newQ;
G.questions = room.questions;
clearInterval(G.timerInterval);
loadQuestion();
showScreen('screen-question');
} else if (phase === 'between' && document.getElementById('screen-question').classList.contains('active')) {
showBetween();
} else if (phase === 'final') {
clearListeners();
showFinal();
}
});
G.listeners.push({ r, unsub });
}

}

G.timerLeft -= 0.1;
if (G.timerLeft <= 0) {
G.timerLeft = 0;
clearInterval(G.timerInterval);
if (!G.answered) timeUp();
}
updateTimer();
}, 100);
}

function updateTimer() {
const pct = Math.max(0, (G.timerLeft / G.timeLimit) * 100);
const fill = document.getElementById('timer-fill');
document.getElementById('timer-num').textContent = Math.ceil(G.timerLeft);
fill.style.width = pct + '%';
fill.className = 'timer-fill' + (pct < 25 ? ' danger' : pct < 50 ? ' warn' : '');
}

async function selectAnswer(idx, btn) {
if (G.answered) return;
G.answered = true;
clearInterval(G.timerInterval);
const elapsed = Math.min((Date.now() - G.answerStart) / 1000, G.timeLimit);
const q = G.questions[G.currentQ];
const correct = idx === q.answer;
const pts = correct ? 200 + Math.round((1 - elapsed/G.timeLimit) * 800) : 0;
document.querySelectorAll('.opt').forEach((b,i) => {
b.disabled = true;
if (i === q.answer) b.classList.add('correct');
else if (i === idx && !correct) b.classList.add('wrong');
});
btn.classList.add('selected');
await recordAnswer(pts, correct, elapsed);
showFeedback(correct, pts, elapsed, q.options[q.answer]);
if (G.isHost) setTimeout(() => showBetween(), 2000);
}

async function timeUp() {
G.answered = true;
const q = G.questions[G.currentQ];
document.querySelectorAll('.opt').forEach((b,i) => {
b.disabled = true;
if (i === q.answer) b.classList.add('correct');
});
await recordAnswer(0, false, G.timeLimit);
showFeedback('timeout', 0, G.timeLimit, q.options[q.answer]);
if (G.isHost) setTimeout(() => showBetween(), 2000);
}

async function recordAnswer(pts, correct, time) {
const pRef = ref(db, `rooms/${G.roomCode}/players/${G.playerName}`);
const snap = await get(pRef);
const cur = snap.val() || { score:0, correct:0, total:0, avgTime:0 };
const newTotal = cur.total + 1;
const newAvg = ((cur.avgTime * cur.total) + time) / newTotal;
await update(pRef, {
score: cur.score + pts,
correct: cur.correct + (correct ? 1 : 0),
total: newTotal,
avgTime: Math.round(newAvg * 10) / 10
});
}

function showFeedback(correct, pts, time, correctAns) {
const el = document.getElementById('feedback');
el.style.display = 'block';
if (correct === 'timeout') {
el.className = 'feedback timeout';
el.innerHTML = `<div style="font-size:20px">⏱ Time's up!</div><div class="msg">Correct answer: <strong>${correctAns}</strong></div>`;
} else if (correct) {
el.className = 'feedback correct';
el.innerHTML = `<div class="pts">+${pts} pts</div><div class="msg">Correct! Answered in ${time.toFixed(1)}s</div>`;
} else {
el.className = 'feedback wrong';
el.innerHTML = `<div style="font-size:20px">✗ Wrong</div><div class="msg">Correct answer: <strong>${correctAns}</strong></div>`;
}
}

async function showBetween() {
const snap = await get(ref(db, `rooms/${G.roomCode}/players`));
const players = snap.val() || {};
const sorted = Object.entries(players).sort((a,b) => b[1].score - a[1].score);
const list = document.getElementById('between-list');
list.innerHTML = '';
sorted.forEach(([name, data], i) => {
const row = document.createElement('div');
row.className = 'score-row';
const medal = i===0?'🥇':i===1?'🥈':i===2?'🥉':`${i+1}.`;
row.innerHTML = `<span class="rank">${medal}</span><span class="sname">${name}${name===G.playerName?' <span style="color:#888;font-weight:400">(you)</span>':''}</span><span class="spts">${data.score} pts</span>`;
list.appendChild(row);
});
const isLast = G.currentQ >= G.questions.length - 1;
document.getElementById('host-next').style.display = G.isHost ? 'block' : 'none';
document.getElementById('guest-between-wait').style.display = G.isHost ? 'none' : 'block';
if (G.isHost) {
const btn = document.getElementById('next-btn');
btn.textContent = isLast ? 'See final results' : 'Next question';
if (isLast) btn.onclick = async () => {
await update(ref(db, `rooms/${G.roomCode}`), { phase: 'final' });
showFinal();
};
await update(ref(db, `rooms/${G.roomCode}`), { phase: 'between' });
}
showScreen('screen-between');
}

window.hostNext = async function() {
G.currentQ++;
await update(ref(db, `rooms/${G.roomCode}`), { currentQ: G.currentQ, phase: 'question' });
loadQuestion();
showScreen('screen-question');
};

async function showFinal() {
clearListeners();
const snap = await get(ref(db, `rooms/${G.roomCode}/players`));
const players = snap.val() || {};
const sorted = Object.entries(players).sort((a,b) => b[1].score - a[1].score);
const winner = sorted[0];
const banner = document.getElementById('winner-banner');
if (winner[0] === G.playerName) {
banner.innerHTML = `<div class="trophy">🏆</div><div class="wname">You won!</div><div class="wsub">First place with ${winner[1].score} points</div>`;
} else {
banner.innerHTML = `<div class="trophy">🎉</div><div class="wname">${winner[0]} wins!</div><div class="wsub">With ${winner[1].score} points</div>`;
}
const list = document.getElementById('final-list');
list.innerHTML = '';
sorted.forEach(([name, data], i) => {
const row = document.createElement('div');
row.className = 'score-row';
const medal = i===0?'🥇':i===1?'🥈':i===2?'🥉':`${i+1}.`;
row.innerHTML = `<span class="rank">${medal}</span><span class="sname">${name}${name===G.playerName?' <span style="color:#888;font-weight:400">(you)</span>':''}<br><span class="detail">${data.correct}/${data.total} correct</span></span><span class="spts">${data.score} pts</span>`;
list.appendChild(row);
});
const me = players[G.playerName] || {};
const myRank = sorted.findIndex(([n])=>n===G.playerName)+1;
document.getElementById('my-stats').innerHTML = `
<div class="stat"><div class="val">${me.score||0}</div><div class="lbl">Total points</div></div>
<div class="stat"><div class="val">${me.correct||0}/${me.total||0}</div><div class="lbl">Correct answers</div></div>
<div class="stat"><div class="val">${me.avgTime||0}s</div><div class="lbl">Avg response time</div></div>
<div class="stat"><div class="val">#${myRank}</div><div class="lbl">Final rank</div></div>
`;
showScreen('screen-final');
}
